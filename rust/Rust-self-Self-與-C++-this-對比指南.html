    }
    
    T distance_to(const Point<T>& other) const {
        T dx = this->x - other.x;
        T dy = this->y - other.y;
        return std::sqrt(dx * dx + dy * dy);
    }

        T this_dist = this->distance_to(Point<T>(0, 0));
        T other_dist = other.distance_to(Point<T>(0, 0));
        
        if (this_dist < other_dist) return -1;
        if (this_dist > other_dist) return 1;
        return 0;
    }
};
