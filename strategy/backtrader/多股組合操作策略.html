            return size * price * self.p.commission
        elif size < 0:  # è³£å‡ºï¼Œè€ƒæ…®å°èŠ±ç¨…
            return -size * price * (self.p.stamp_duty + self.p.commission)
        else:
            return 0  # just in case for some reason the size is 0.
